19  marked_ptr(T* p = nullptr, uintptr_t mark = 0) noexcept
20  {
21  assert(mark <= MarkMask && "mark exceeds the number of bits reserved");
22  assert((reinterpret_cast<uintptr_t>(p) & MarkMask) == 0 &&
23  "bits reserved for masking are occupied by the pointer");
24  ptr = reinterpret_cast<T*>(reinterpret_cast<uintptr_t>(p) | mark);
25  }
26

32  return reinterpret_cast<uintptr_t>(ptr) & MarkMask;
33  }

35  // Get underlying pointer (with mark bits stripped off).
36  T* get() const noexcept {
37  return reinterpret_cast<T*>(reinterpret_cast<uintptr_t>(ptr) & ~MarkMask);
38  }

41  explicit operator bool() const noexcept { return ptr != nullptr; }
